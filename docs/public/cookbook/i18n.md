# Cookbook: Internationalization (i18n)

Add multi-language support with locale switching and RTL layout support.

## Setup

Create a translations module at `src/i18n.js`:

```js
import { createSignal, createMemo } from 'decantr/state';
import { createPersisted } from 'decantr/data';

// Persist the user's locale choice
const [locale, setLocale] = createPersisted('app-locale', 'en');

// Message dictionaries
const messages = {
  en: {
    nav: { home: 'Home', about: 'About', settings: 'Settings', logout: 'Logout' },
    home: { title: 'Welcome', subtitle: 'Build something great today.' },
    auth: { login: 'Sign In', register: 'Create Account', email: 'Email', password: 'Password', forgot: 'Forgot password?' },
    common: { save: 'Save', cancel: 'Cancel', delete: 'Delete', edit: 'Edit', loading: 'Loading...', search: 'Search...' },
    users: { title: 'Users', addUser: 'Add User', name: 'Name', role: 'Role', status: 'Status' },
  },
  es: {
    nav: { home: 'Inicio', about: 'Acerca de', settings: 'Configuracion', logout: 'Cerrar sesion' },
    home: { title: 'Bienvenido', subtitle: 'Construye algo genial hoy.' },
    auth: { login: 'Iniciar sesion', register: 'Crear cuenta', email: 'Correo', password: 'Contrasena', forgot: 'Olvidaste tu contrasena?' },
    common: { save: 'Guardar', cancel: 'Cancelar', delete: 'Eliminar', edit: 'Editar', loading: 'Cargando...', search: 'Buscar...' },
    users: { title: 'Usuarios', addUser: 'Agregar usuario', name: 'Nombre', role: 'Rol', status: 'Estado' },
  },
  fr: {
    nav: { home: 'Accueil', about: 'A propos', settings: 'Parametres', logout: 'Deconnexion' },
    home: { title: 'Bienvenue', subtitle: 'Construisez quelque chose de genial.' },
    auth: { login: 'Connexion', register: 'Creer un compte', email: 'Email', password: 'Mot de passe', forgot: 'Mot de passe oublie ?' },
    common: { save: 'Enregistrer', cancel: 'Annuler', delete: 'Supprimer', edit: 'Modifier', loading: 'Chargement...', search: 'Rechercher...' },
    users: { title: 'Utilisateurs', addUser: 'Ajouter un utilisateur', name: 'Nom', role: 'Role', status: 'Statut' },
  },
  ar: {
    nav: { home: 'الرئيسية', about: 'حول', settings: 'الإعدادات', logout: 'تسجيل الخروج' },
    home: { title: 'مرحباً', subtitle: 'ابنِ شيئاً رائعاً اليوم.' },
    auth: { login: 'تسجيل الدخول', register: 'إنشاء حساب', email: 'البريد الإلكتروني', password: 'كلمة المرور', forgot: 'نسيت كلمة المرور؟' },
    common: { save: 'حفظ', cancel: 'إلغاء', delete: 'حذف', edit: 'تعديل', loading: 'جار التحميل...', search: 'بحث...' },
    users: { title: 'المستخدمون', addUser: 'إضافة مستخدم', name: 'الاسم', role: 'الدور', status: 'الحالة' },
  },
};

// RTL locales
const rtlLocales = new Set(['ar', 'he', 'fa', 'ur']);

// Translation function — resolves dot-notation keys like 'nav.home'
function t(key) {
  const lang = locale();
  const dict = messages[lang] || messages.en;
  const parts = key.split('.');
  let result = dict;
  for (const part of parts) {
    result = result?.[part];
    if (result === undefined) {
      // Fallback to English
      let fallback = messages.en;
      for (const p of parts) {
        fallback = fallback?.[p];
      }
      return fallback || key;
    }
  }
  return result;
}

// Reactive direction
const direction = createMemo(() => rtlLocales.has(locale()) ? 'rtl' : 'ltr');

// Available locales for UI display
const availableLocales = [
  { value: 'en', label: 'English' },
  { value: 'es', label: 'Espanol' },
  { value: 'fr', label: 'Francais' },
  { value: 'ar', label: 'العربية' },
];

export { locale, setLocale, t, direction, availableLocales };
```

## Applying Direction to the Document

In `src/app.js`, update the document direction when the locale changes:

```js
import { createEffect } from 'decantr/state';
import { direction } from './i18n.js';

createEffect(() => {
  document.documentElement.dir = direction();
  document.documentElement.lang = locale();
});
```

## Locale Switcher Component

```js
import { tags } from 'decantr/tags';
import { css } from 'decantr/css';
import { Select } from 'decantr/components';
import { locale, setLocale, availableLocales } from '../i18n.js';

const { div } = tags;

export function LocaleSwitcher() {
  return Select({
    value: locale,
    options: availableLocales,
    onchange: (val) => setLocale(val),
    class: css('_w[160px]')
  });
}
```

## Using Translations in Pages

```js
import { tags } from 'decantr/tags';
import { text } from 'decantr/core';
import { css } from 'decantr/css';
import { Button, Input } from 'decantr/components';
import { t } from '../i18n.js';
import { LocaleSwitcher } from '../components/locale-switcher.js';

const { div, h1, p, header, main } = tags;

export default function HomePage() {
  return div({ class: css('_flex _col _minhscreen') },

    header({ class: css('_flex _aic _jcsb _px6 _py3 _borderB') },
      div({ class: css('_flex _gap4') },
        text(() => t('nav.home')),
        text(() => t('nav.about')),
        text(() => t('nav.settings'))
      ),
      LocaleSwitcher()
    ),

    main({ class: css('_flex _col _aic _jcc _flex1 _gap4 _p8') },
      h1({ class: css('_heading1 _tc') }, text(() => t('home.title'))),
      p({ class: css('_body _fgmuted _tc') }, text(() => t('home.subtitle'))),
      div({ class: css('_flex _gap3') },
        Input({ placeholder: () => t('common.search') }),
        Button({ variant: 'primary' }, text(() => t('common.save')))
      )
    )
  );
}
```

Since `t()` reads the reactive `locale()` signal, all text updates automatically when the user switches languages.

## RTL-Aware Layout with Logical Property Atoms

Decantr provides logical property atoms that work correctly in both LTR and RTL contexts:

```js
// Physical properties (DON'T use for RTL support)
css('_pl4')    // padding-left — stays left in RTL
css('_pr4')    // padding-right — stays right in RTL

// Logical properties (DO use for RTL support)
css('_pis4')   // padding-inline-start — becomes right in RTL
css('_pie4')   // padding-inline-end — becomes left in RTL
css('_mis4')   // margin-inline-start
css('_mie4')   // margin-inline-end
```

### RTL-Safe Layout Example

```js
div({ class: css('_flex _aic _gap4 _pis6 _pie4') },
  // In LTR: padding-left: 1.5rem, padding-right: 1rem
  // In RTL: padding-right: 1.5rem, padding-left: 1rem
  Avatar({ name: 'Ali', size: 'sm' }),
  div({ class: css('_flex _col _flex1') },
    span({ class: css('_bold') }, 'Ali Hassan'),
    span({ class: css('_textsm _fgmuted') }, 'Developer')
  ),
  Button({ variant: 'ghost', size: 'sm' }, icon('chevron-right'))
)
```

## Interpolated Messages

For messages with dynamic values, use template functions:

```js
const messages = {
  en: {
    users: {
      greeting: (name) => `Hello, ${name}!`,
      count: (n) => `${n} user${n === 1 ? '' : 's'} found`,
    }
  },
  es: {
    users: {
      greeting: (name) => `Hola, ${name}!`,
      count: (n) => `${n} usuario${n === 1 ? '' : 's'} encontrado${n === 1 ? '' : 's'}`,
    }
  }
};
```

Usage:

```js
span({}, text(() => t('users.greeting')('Alice')))
span({}, text(() => t('users.count')(users.data()?.length || 0)))
```

## Date and Number Formatting

Use the browser's built-in `Intl` APIs with the reactive locale:

```js
import { createMemo } from 'decantr/state';
import { locale } from '../i18n.js';

const dateFormatter = createMemo(() => new Intl.DateTimeFormat(locale(), {
  year: 'numeric', month: 'long', day: 'numeric'
}));

const numberFormatter = createMemo(() => new Intl.NumberFormat(locale(), {
  style: 'currency', currency: 'USD'
}));

// Usage in a component
span({}, text(() => dateFormatter().format(new Date())))
span({}, text(() => numberFormatter().format(1234.56)))
// en: "March 16, 2026" / "$1,234.56"
// ar: "١٦ مارس ٢٠٢٦" / "١٬٢٣٤٫٥٦ US$"
```

## Key Takeaways

- Keep translations in a single module that exports `t()`, `locale`, and `setLocale`
- `t()` reads a reactive signal, so all text updates automatically on locale change
- Use `createPersisted` to remember the user's language preference
- Use logical property atoms (`_pis`, `_pie`, `_mis`, `_mie`) for RTL-safe layouts
- Set `document.documentElement.dir` and `lang` reactively for proper browser behavior
- Lean on `Intl.DateTimeFormat` and `Intl.NumberFormat` for locale-aware formatting
